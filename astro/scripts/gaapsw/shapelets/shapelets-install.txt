#INSTALLATION INSTRUCTIONS FOR THE PSF GAUSSIANIZATION ROUTINES

#(this also builds cfitsio and pgplot libraries).

cd $GAaP

cd cfitsio
./configure
make clean
make
cp libcfitsio.a ../shapelets/lib/

cd ../shapelets/lib
rm *.o libpgplot.* grfont.dat pgxwin_server 
make all
mv libpgplot.so libpgplot.so.hide

cd ../kk
make clean
make set

cd bigim
make clean
make set

#in the shell from which you run the software you need to set two
#environment variables:

setenv PGPLOT_DIR $GAaP/shapelets/lib/
setenv PGPLOT_FONT $PGPLOT_DIR/grfont.dat
